#include "DHT.h"

#define PIN_LDR_IN      13
#define PIN_DHT_DATA    14
#define PIN_DUST_LED    5
#define PIN_DUST_IN     39

#define PIN_LED_DHT     21
#define PIN_LED_LDR     22
#define PIN_LED_DUST    23

DHT dht(PIN_DHT_DATA, DHT11);

unsigned long previousTime = 0;
const long interval = 1000;

void setup() {
  Serial.begin(115200);
  dht.begin();

  pinMode(PIN_LED_DHT, OUTPUT);
  pinMode(PIN_LED_LDR, OUTPUT);
  pinMode(PIN_LED_DUST, OUTPUT);

  pinMode(PIN_LDR_IN, INPUT);
  pinMode(PIN_DUST_LED, OUTPUT);
  pinMode(PIN_DUST_IN, INPUT);
}

void loop() {
  unsigned long currentTime = millis();

  // --- 1. TIMEOUT THOI GIAN ---
  if (currentTime - previousTime >= interval) {
    previousTime = currentTime;


    // --- 2. DOC DU LIEU CAM BIEN ---
    float t = dht.readTemperature();
    float h = dht.readHumidity();
    
    int ldrVal = analogRead(PIN_LDR_IN);

    digitalWrite(PIN_DUST_LED, LOW);
    delayMicroseconds(280);
    int dustRaw = analogRead(PIN_DUST_IN);
    delayMicroseconds(40);
    digitalWrite(PIN_DUST_LED, HIGH);
    delayMicroseconds(9680);

    float voltage = dustRaw * (3.3 / 4095.0);
    float dustVal = (0.17 * voltage - 0.1) * 1000;
    if (dustVal < 0) dustVal = 0;


    // --- 3. DIEU KHIEN DEN (LED) ---
    digitalWrite(PIN_LED_DHT, !isnan(t) ? HIGH : LOW);
    digitalWrite(PIN_LED_LDR, ldrVal > 100 ? HIGH : LOW);
    digitalWrite(PIN_LED_DUST, dustVal > 0 ? HIGH : LOW);


    Serial.print("T: "); Serial.print(t);
    Serial.print(" | H: "); Serial.print(h);
    Serial.print(" | L: "); Serial.print(ldrVal);
    Serial.print(" | D: "); Serial.println(dustVal);
  }
}
